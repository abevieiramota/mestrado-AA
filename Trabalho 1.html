PATH_DADOS = "Dados Exercicio 1"
FILENAMES = ["ex1data1.txt", "ex1data2.txt", "ex1data3.txt"]


def carregar(filename, delimiter=" "):
    import csv
    import os
    
    reader = csv.reader(open(os.path.join(PATH_DADOS, filename)), delimiter=delimiter)
    
    values_X = []
    values_Y = []

    for row in reader:
        values_X.append([float(x) for x in row[:-1]])
        values_Y.append(float(row[-1]))
    
    import numpy as np

    return np.array(values_X), np.array(values_Y)

def plot2d(X, Y, color=None):
    from matplotlib import pyplot as pl
        
    line = pl.plot(X, Y)
    if not color is None:
        pyplot.setp(line, color=color)

def scatter2d(X, Y):
    from matplotlib import pyplot as pl
    
    pl.scatter(X, Y)

Ela envolve aspectos como a preciso que se deseja alcanar com o modelo e quo prximo dos dados de treinamento espera-se que os dados a serem predizidos sejam. Pelo grfico, os dados no parecem possuir uma boa aproximao por uma reta, por possuir, para valores prximos de x, variaes no monotnicas de y.

# retorna o EQM, a mdia dos erros e a mdia dos erros pelos x's
def eqm(x, gerado, esperado):
    resultado = 0.
    
    for g, e in zip(gerado, esperado):
        erro = g - e
        erro_quadrado = erro**2
        
        resultado += erro_quadrado
        
    resultado /= len(gerado)
    
    return resultado

def eq_1grau(w0, w1, x):
    
    return w0 + w1*x
    

def grad_desc_estoq_univ(X, Y, alfa=0.1, epocas=100, w0 = None, w1 = None, plot=False):
    # valores randomicos para w0 e w1
    if w0 is None:
        w0 = random.uniform(1, 10)
    if w1 is None:
        w1 = random.uniform(1, 10)
    
    eqms = []
    
    # percorre os valores de x e y
    from itertools import cycle
    xs = cycle(X)
    ys = cycle(Y)
    
    for epoca, x, y in zip(xrange(epocas), xs, ys):
        # calcula as predies para os dados de treinamento
        predicao = [eq_1grau(w0, w1, elem[0]) for elem in X]
        
        # calcula o eqm
        eqm_calculado = eqm(X, predicao, Y)
        eqms.append(eqm_calculado)
        
        # erro da iterao
        erro = y - eq_1grau(w0, w1, x[0])
        # atualiza os pesos
        w0 += alfa*erro
        w1 += alfa*erro*x[0]
        
    if plot:
        scatter2d(X, Y)
        plot2d(X, [eq_1grau(w0, w1, x[0]) for x in X], color="r")
        
    return w0, w1, eqms

Comentários: Através do gráfico “épocas x EQM” é possível verificar que o algoritmo está “aprendendo” ? Comente.¶

Considerando que uma diminuio dos erros do modelo em predies sobre dados do training-set representam aprendizado, pode-se sim afirmar que o algoritmo est "aprendendo". Podemos pensar cada poca de treinamento como um incremento no aprendizado.
